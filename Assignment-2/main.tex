\textbf{Exercise 12.13.1.10} A black and a red dice are rolled.
\begin{enumerate}[label=(\alph*)]
    \item Find the conditional probability of obtaining a sum greater than 9, given
that the black die resulted in a 5.
    \item Find the conditional probability of obtaining the sum 8, given that the red die
resulted in a number less than 4.
\end{enumerate}

\textbf{Solution.} Let $X$ and $Y$ be the random variables denoting the number which comes up on black and red die respectively.

\begin{align}
\label{prdice}
\pr{X= n } = \pr{Y = n} = 
\begin{cases}
\frac{1}{6} & 1 \le n \le 6
\\
0 & otherwise
\end{cases}
\end{align}

Let $Z = X + Y$,
\begin{align}
\pr{Z= n} &= 
\begin{cases}
0 & n \le 1
\\
\frac{n-1}{36} &  2 \le n \le  7
\\
\frac{13-n}{36} & 7 < n \le 12
\\
0 & n > 12
\end{cases}
\label{prsum}
\end{align}


Also, conditional probability of $X>r$ given 
$Y= k$  is:
\begin{align}
    \pr{X>r | Y=k} = \frac{\pr{X>r, Y=k}}{\pr{Y=k}}
\end{align}

$\because$Rolling of black and red die is independent of each other,
\begin{align}
    \therefore \label{eq2}\pr{X=r , Y=k} &= \pr{X=r}\pr{Y= k}\\
    \pr{Z= k , X = r} &= \pr{Y = k -r ,X = r}\\
                      &= \pr{Y= k-r}\pr{X =r}
\end{align}
\begin{align}
     \therefore \label{zx} 
            \pr{Z= k , X = r} = 
            \begin{cases}
                \frac{1}{36} &   0 <r \le 6, 0<k -r \le 6 ,\\
                0 & otherwise
            \end{cases} 
\end{align}

 WLOG, $Z$ and $Y$ follow,

\begin{align}
    \therefore \label{zy} 
            \pr{Z= k , Y = r} = 
            \begin{cases}
                \frac{1}{36} &   0 <r \le 6, 0<k-r\le 6 \\
                0 & otherwise
            \end{cases} 
\end{align}

\begin{enumerate}[label=(\alph*)]
    \item here we need to find,
    \begin{align}
        \pr{Z> 9 | X = 5} &= \frac{\pr{Z> 9, X = 5}}{\pr{X = 5}}\\
                              &= \frac{\sum_{i = 10}^{12}\pr{X=5, Z = i}}{\pr{X = 5}}  
    \end{align}

    from $\eqref{zx}$
    \begin{align}
        \pr{Z> 9 | X = 5}     &= \frac{\frac{1}{36} + \frac{1}{36} +0}{\frac{1}{6} }\\
                              &= \frac{2}{6}\\
                              &= \frac{1}{3} \approx 0.33
    \end{align}

\item here, we need to find,
\begin{align}
\pr{Z = 8 | Y < 4} &= \frac{\pr{Z = 8, Y < 4}}{\pr{Y < 4}}\\
                       &= \frac{\sum_{i = 1}^{3}\pr{Z = 8, Y = i}}{\sum_{i = 1}^{3}\pr{Y = i}}
\end{align}
from $\eqref{zy}$

\begin{align}
    \pr{Z = 8 | Y < 4} &= \frac{ 0 + \frac{1}{36}+\frac{1}{36}}{\frac{1}{6} +\frac{1}{6} +\frac{1}{6}}\\
    &= \frac{1}{9} \approx 0.11
\end{align}
\end{enumerate}
